-> Cte bảng tạm là kỹ thuật tối ưu trong SQL nên dùng khi không thể viết khéo để query bảng 1 lần. Nguyên tắc là kbh được viết query cùng 1 thứ nhiều lần mà phải tối ưu. VD:
WITH AppTemp AS (
  SELECT PowerAppId, IsSubmitted, ModifiedTime, CreateTime
  FROM [xxxxxxxxxxxxxxx].[PowerAppRenewNotificationFeedbackVerify]
  WHERE PowerAppId IN ('xxxxxxxxxxxxxxxxxxxx')
)
SELECT Y.PowerAppId FROM 
(SELECT PowerAppId, ModifiedTime FROM AppTemp WHERE IsSubmitted = 1) as Y
INNER JOIN 
(SELECT Max(Greatest(CreateTime, ModifiedTime)) as maxTime, PowerAppId FROM AppTemp GROUP BY PowerAppId) as X
On X.PowerAppId = Y.PowerAppId AND X.maxTime = Y.ModifiedTime
